1. Prerequisites

Verification of the following environment settings is required for a successful build:

  • Xcode 15.0+ Required
  • Deployment Target: iOS 16.1+
  • Swift 5.9+

2. API & Required Parameters

To initialize the SDK, you must have the following configuration parameters ready:

● Google Maps API Key: Required for location services.
● User Mobile Number: Session authentication.

3. Framework Setup

Place the FlutterSDK folder containing the .podspec file in your project's root directory

Project-Root/
├── FlutterSDK/ <-- Flutter SDK folder
│ ├── FlutterSDK.podspec
│ └── Frameworks/
│ └── *.xcframework files
├── ShoppingGateSDKDemo/
├── Podfile
├── Podfile.lock
├── Pods/
├── ShoppingGateSDKDemo.xcodeproj
└── ShoppingGateSDKDemo.xcworkspace
Local Pod: The SDK is integrated as a local CocoaPod using the .podspec file. Use same folder structure for Flutter module XCFrameworks in the FlutterSDK/Frameworks/ directory.

4. Podfile Configuration

Reference the local ShoppingGateSDK.podspec in your Podfile

Podfile
platform :ios, '16.1'

target 'YourApp' do
  use_frameworks!

  # Local ShoppingGate SDK (includes Flutter engine and dependencies)
  pod 'FlutterSDK', :path => './FlutterSDK'
end

Example ShoppingGateSDK.podspec

ShoppingGateSDK/ShoppingGateSDK.podspec

Pod::Spec.new do |s|
  s.name             = 'FlutterSDK'
  s.version          = '1.0.0'
  s.summary          = 'High-performance Flutter module for native iOS integration.'
  s.description      = 'This SDK bundles the Flutter engine and your custom module logic.'
  s.homepage         = 'https://yourcompany.com'
  s.license          = { :type => 'MIT', :file => 'LICENSE' }
  s.author           = { 'Your Team' => 'dev@yourcompany.com' }
  s.source           = { :path => '.' } # Points to the local folder

  s.ios.deployment_target = '16.1'

  # 1. Link your Flutter frameworks
  # This tells CocoaPods to include all xcframeworks inside the Frameworks folder
  s.vendored_frameworks = 'Frameworks/*.xcframework'

  # 2. Declare External Dependencies
  # This fixes the "Undefined Symbol" errors by downloading the actual SDKs
  s.dependency 'FirebaseAnalytics'
  s.dependency 'GoogleMaps'

  # 3. Ensure the Linker Flag is set
  s.pod_target_xcconfig = { 'OTHER_LDFLAGS' => '-ObjC' }
  
  s.static_framework = true
end

After updating Podfile, run:

Terminal
pod install
Pro Tip: Using .podspec allows automatic dependency management. All required Flutter plugins and frameworks are resolved automatically through the podspec file.

5. Info.plist & Permissions

Add required permissions and Google Maps API Key to Info.plist

Info.plist
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
    <!-- Google Maps API Key -->
    <key>GMSApiKey</key>
    <string>YOUR_MAPS_KEY_HERE</string>
    
    <!-- Location Permissions -->
    <key>NSLocationWhenInUseUsageDescription</key>
    <string>We need your location to show nearby stores</string>
    
    <key>NSLocationAlwaysAndWhenInUseUsageDescription</key>
    <string>We need your location to show nearby stores</string>
    
    <!-- Camera & Photo Library -->
    <key>NSCameraUsageDescription</key>
    <string>We need camera access to capture product photos</string>
    
    <key>NSPhotoLibraryUsageDescription</key>
    <string>We need photo library access to select images</string>
    
    <key>NSPhotoLibraryAddUsageDescription</key>
    <string>We need permission to save photos</string>
</dict>
</plist>

6. Launching the SDK

Implement FlutterViewController to launch the ShoppingGate SDK with user authentication

A. AppDelegate Setup

AppDelegate.swift
import UIKit
import Flutter
import GoogleMaps

@main
class AppDelegate: UIResponder, UIApplicationDelegate {
    
    lazy var flutterEngine = FlutterEngine(name: "ShoppingGateEngine")
    
    func application(
        _ application: UIApplication,
        didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?
    ) -> Bool {
        
        // Initialize Google Maps
        GMSServices.provideAPIKey("YOUR_MAPS_KEY_HERE")
        
        // Pre-warm Flutter engine for faster startup
        flutterEngine.run()
        
        return true
    }
}

B. ViewController Implementation

ViewController.swift
import UIKit
import Flutter

class ViewController: UIViewController {
    
    override func viewDidLoad() {
        super.viewDidLoad()
        
        // Setup Launch Button
        let launchButton = UIButton(type: .system)
        launchButton.setTitle("Launch ShoppingGate", for: .normal)
        launchButton.titleLabel?.font = .systemFont(ofSize: 18, weight: .semibold)
        launchButton.backgroundColor = .systemIndigo
        launchButton.setTitleColor(.white, for: .normal)
        launchButton.layer.cornerRadius = 12
        launchButton.translatesAutoresizingMaskIntoConstraints = false
        launchButton.addTarget(self, action: #selector(launchSDK), for: .touchUpInside)
        
        view.addSubview(launchButton)
        
        NSLayoutConstraint.activate([
            launchButton.centerXAnchor.constraint(equalTo: view.centerXAnchor),
            launchButton.centerYAnchor.constraint(equalTo: view.centerYAnchor),
            launchButton.widthAnchor.constraint(equalToConstant: 240),
            launchButton.heightAnchor.constraint(equalToConstant: 50)
        ])
    }
    
    @objc func launchSDK() {
        guard let appDelegate = UIApplication.shared.delegate as? AppDelegate else {
            return
        }
        
        // Create Flutter view controller with initial route
        let flutterViewController = FlutterViewController(
            engine: appDelegate.flutterEngine,
            nibName: nil,
            bundle: nil
        )
        
        navigationController?.pushViewController(flutterViewController, animated: true)
    }
}
